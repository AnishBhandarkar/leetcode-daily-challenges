1840. Maximum Building Height


/**
 * @param {number} n
 * @param {number[][]} restrictions
 * @return {number}
 */
var maxBuilding = function (n, restrictions) {
    // Add the start and end boundary conditions
    restrictions.push([1, 0]);
    restrictions.push([n, n - 1]);

    // sort by building position
    restrictions.sort((a, b) => a[0] - b[0]);

    let m = restrictions.length;

    // left to right pass
    for (let i = 1; i < m; i++) { // No need to include 1st building
        let distance = restrictions[i][0] - restrictions[i - 1][0];
        let maxPossibleHeight = restrictions[i - 1][1] + distance;
        restrictions[i][1] = Math.min(restrictions[i][1], maxPossibleHeight);
    }

    // right to left pass
    for (let i = m - 2; i >= 0; i--) { // No need to include last building
        let distance = restrictions[i + 1][0] - restrictions[i][0];
        let maxPossibleHeight = restrictions[i + 1][1] + distance;
        restrictions[i][1] = Math.min(restrictions[i][1], maxPossibleHeight);
    }

    // Find the highest peak between adjacent restricted buildings
    // i.e (h1 + h2 + distance) / 2

    let maxHeight = 0;
    for (let i = 1; i < m; i++) {
        let pos1 = restrictions[i - 1][0];
        let pos2 = restrictions[i][0];
        let distance = pos2 - pos1;

        let h1 = restrictions[i - 1][1];
        let h2 = restrictions[i][1];

        let peak = Math.trunc((h1 + h2 + distance) / 2);
        maxHeight = Math.max(maxHeight, peak);
    }
    return maxHeight;
};


---


Complexity:

Time complexity: O(m log m) for sorting, where m is the number of restrictions after adding the boundary conditions. The two linear passes are O(m). Overall O(m log m).

Space complexity: O(1) extra space beyond the input array (in-place updates to restrictions). If you count the input array, it is O(m); otherwise, auxiliary space is O(1).
