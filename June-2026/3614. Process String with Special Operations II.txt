// Appraoch 1: {Lead to Memory Limit Exceeded (MLE)}

T(n) = O(n^2)
S(n) = O(n)


/**
 * @param {string} s
 * @param {number} k
 * @return {character}
 */
var processStr = function (s, k) {
    let result = [];

    let specialChars = new Set(['*', '#', '%']);

    for (let ch of s) {
        if (!specialChars.has(ch)) {
            result.push(ch);
        } else {
            if (ch === '*' && result.length > 0) {
                result.pop();
            } else if (ch === '#') {
                result.push(...result);
            } else if (ch === '%') {
                result.reverse();
            }
        }
    }

    return k >= result.length ? "." : result[k];
};

// Approach 2:


> So the algorithm is:
- Compute the final length.
- Move through operations in reverse.
- Keep translating k to the position it had before each operation.
- Eventually you'll reach the original character that generated that position.



/**
 * @param {string} s
 * @param {number} k
 * @return {character}
 */
var processStr = function (s, k) {
    // Compute the final length without actually building the string.
    let len = 0;

    for (const ch of s) {
        if (ch === '*') {
            // Delete the last character if the string is non-empty.
            len = len > 0 ? len - 1 : 0;
        } else if (ch === '#') {
            // Duplicate the current string.
            len *= 2;
        } else if (ch === '%') {
            // Reverse does not change the length.
            continue;
        } else {
            // Append a normal character.
            len++;
        }
    }

    // k is outside the final string.
    if (k >= len) return ".";

    // Walk backwards and undo each operation.
    for (let i = s.length - 1; i >= 0; i--) {
        const ch = s[i];

        if (ch === '*') {
            // Forward: removed one character.
            // Backward: restore the deleted position.
            len++;
        } else if (ch === '#') {
            // Forward: S -> S + S
            // If idx is in the second half, map it to the first half.
            len /= 2n;

            if (k >= len) {
                k -= len;
            }
        } else if (ch === '%') {
            // Forward: reverse the string.
            // Backward: reverse again.
            k = len - 1 - k;
        } else {
            // Forward: appended this character at position len - 1.
            len--;

            // If idx points to that appended position,
            // we've found the answer.
            if (k === len) {
                return ch;
            }
        }
    }

    return ".";
};

/*
Time:  O(n)
Space: O(1)
*/
